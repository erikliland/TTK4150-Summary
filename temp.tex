\section{Definitions of stability}
\paragraph{Stability} The origin is stable iff
\begin{equation}
	\forall \quad \epsilon > 0 \quad \exists \quad \delta(\epsilon) > 0
	\mbox{ s.t. }
	||x(0)|| < \delta \implies ||x(t)|| < \epsilon \quad \forall \quad t \geq 0
\end{equation}
