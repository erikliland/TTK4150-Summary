\section{Backstepping}

\paragraph{Method}
Given a system
\begin{align}
	\dot{\eta}_1 &= f_1(\eta) + g_1(\eta)\xi \\
	             &\hspace{7pt}\vdots         \\
	\dot{\eta}_n &= f_n(\eta) + g_n(\eta)\xi \\
	\dot{\xi}    &= u
\end{align}

\begin{enumerate}
	\item Find an input $\xi_1$ to stabilize $\dot{\eta}_1 = f(\eta) + g(\eta)\xi_1$
\end{enumerate}
